WordPress + Google Tag Manager — OTP iframe + hidden dealer price until verified
===================================================================================

Behavior
--------
- Masks "AM Ford Price" / `.price-block.our-price.real-price` on each inventory card with a
  teaser image + navy "Unlock Instant Price" button (image URL is configurable below).
- "Unlock Instant Price" and "Get Today's Price" open the OTP iframe (same flow).
- While locked, Dealer Inspire green `.lightning-custom-cta.stat-button-link` ("Get Today's Price") buttons stay hidden — only the navy Unlock UI shows.
- After **one successful OTP**, the shopper is treated as verified for this browser: other vehicles still show the Unlock UI until tapped, but tapping **Unlock Instant Price** only **reveals that card’s price** (no second OTP). The first OTP still unlocks the vehicle you verified on; each other vehicle unlocks on first click after that.
- Legacy green buttons are **never** shown again (no restore after verify).
- While locked, **dealer discount**, **doc/fee**, and similar adjustment rows in the price stack are hidden (along with the blurred dealer price); **the moment that vehicle’s discounted price is revealed**, those rows come back **except** the legacy green “Get Today’s Price” link, which stays removed for good.
- When a card’s price is revealed (OTP success or one-tap unlock after verification), a short **confetti** burst plays on that card (skipped when the visitor prefers reduced motion).

Paste into GTM → Custom HTML. Set POPUP_ORIGIN (no trailing slash). Optional: change PRICE_TEASER_IMAGE_URL for the green banner graphic.

  function unlockKeyFromContext(ctx) {
    if (!ctx) return '';
    var v = (ctx.vin || '').trim().toUpperCase();
    if (v) return 'V:' + v;
    var s = (ctx.stock || '').trim().toUpperCase();
    if (s) return 'S:' + s;
    return '';
  }

  function unlockKeyFromPayload(payload) {
    if (!payload || typeof payload !== 'object') return '';
    if (payload.car) {
      var c = payload.car;
      var v1 = (c.vin || '').trim().toUpperCase();
      if (v1) return 'V:' + v1;
      var s1 = (c.stock || '').trim().toUpperCase();
      if (s1) return 'S:' + s1;
    }
    var v2 = (payload.vin || '').trim().toUpperCase();
    if (v2) return 'V:' + v2;
    var s2 = (payload.stock || '').trim().toUpperCase();
    if (s2) return 'S:' + s2;
    return '';
  }

  function getUnlockedKeys() {
    try {
      var raw = localStorage.getItem(VINS_STORAGE_KEY);
      if (!raw) return [];
      var arr = JSON.parse(raw);
      return Array.isArray(arr) ? arr : [];
    } catch (e) {
      return [];
    }
  }

  function isKeyUnlocked(key) {
    if (!key) return false;
    return getUnlockedKeys().indexOf(key) >= 0;
  }

  function addUnlockedKey(key) {
    if (!key) return;
    try {
      var arr = getUnlockedKeys();
      if (arr.indexOf(key) >= 0) return;
      arr.push(key);
      localStorage.setItem(VINS_STORAGE_KEY, JSON.stringify(arr));
    } catch (e) {}
  }

    try {
      return localStorage.getItem(GLOBAL_VERIFY_KEY) === '1';

    schedulePriceRevealCelebrationForVehicle(vin, stock);
  }

  /** Hide Dealer Inspire green "Get Today's Price" CTAs while our navy Unlock UI is shown */
  function hideLegacyGetPriceCtasInCard(wrap) {
    if (!wrap || !wrap.querySelectorAll) return;
    var nodes = wrap.querySelectorAll('.lightning-custom-cta.stat-button-link');
    for (var i = 0; i < nodes.length; i++) {
      var node = nodes[i];
      var txt = (node.textContent || '').replace(/\s+/g, ' ').trim();
      if (!matchesGetTodaysPriceLabel(txt)) continue;
      node.setAttribute('data-otp-legacy-cta-hidden', '1');
      node.style.setProperty('display', 'none', 'important');
      node.style.setProperty('visibility', 'hidden', 'important');
      node.style.setProperty('pointer-events', 'none', 'important');
    }
  }

  function safeParseDataVehicle(raw) {
    if (!raw || typeof raw !== 'string') return null;
    try {
      return JSON.parse(raw);
    } catch (e1) {
      try {
        return JSON.parse(raw.replace(/&quot;/g, '"'));
      } catch (e2) {
        return null;
      }
    }
  }

  function findVehicleCardRoot(el) {
    if (!el || typeof el.closest !== 'function') return null;
    var cardRoot = el.closest('.result-wrap');
    if (cardRoot) return cardRoot;
    cardRoot = el.closest('[data-vehicle]');
    if (cardRoot && cardRoot.getAttribute('data-vehicle')) return cardRoot;
    return document.querySelector('.result-wrap[data-vehicle]') || document.querySelector('[data-vehicle]');
  }

  /** Best-effort hero photo from the inventory card DOM (Cars Commerce, Dealer Inspire, etc.) */
  function imgEffectiveSrc(img) {
    if (!img) return '';
    try {
      if (img.currentSrc) return String(img.currentSrc).trim();
    } catch (e1) {}
    return (
      img.getAttribute('src') ||
      img.getAttribute('data-src') ||
      img.getAttribute('data-lazy-src') ||
      img.getAttribute('data-original') ||
      ''
    ).trim();
  }

  function isLikelyVehicleHeroPhotoUrl(src) {
    if (!src || typeof src !== 'string') return false;
    var u = src.trim();
    if (!u) return false;
    if (u.indexOf('//') === 0) u = 'https:' + u;
    if (u.indexOf('http://') !== 0 && u.indexOf('https://') !== 0) return false;
    var low = u.toLowerCase();
    if (low.indexOf('vehicle-images.carscommerce') !== -1 || low.indexOf('carscommerce.inc') !== -1)
      return true;
    if (low.indexOf('data:image') === 0) return false;
    if (low.indexOf('am-ford') !== -1 || low.indexOf('amford') !== -1) return false;
    if (low.indexOf('dealerinspire') !== -1 && (low.indexOf('logo') !== -1 || low.indexOf('am-ford') !== -1))
      return false;
    return true;
  }

  function findVehicleCardHeroImageUrl(root) {
    if (!root || !root.querySelectorAll) return '';
    var selectors = [
      'img[data-testid^="vehicle-image"]',
      'img[data-testid="vehicle-image-1"]',
      '[data-testid^="vehicle-image"] img',
      '.result-vehicle-image img',
      '.vehicle-image img',
      '.inventory-image img',
      '.hit-image img',
      'a.hit-link img',
      '.advanced-pricing-vrp-card ~ * img',
      '.result-wrap .primary-vehicle-photo img'
    ];
    var seen = {};
    var i,
      j,
      imgs,
      img,
      src,
      selIdx,
      imgIdx;
    for (selIdx = 0; selIdx < selectors.length; selIdx++) {
      imgs = root.querySelectorAll(selectors[selIdx]);
      for (imgIdx = 0; imgIdx < imgs.length; imgIdx++) {
        img = imgs[imgIdx];
        src = imgEffectiveSrc(img);
        if (!src || seen[src]) continue;
        seen[src] = 1;
        if (isLikelyVehicleHeroPhotoUrl(src)) return src;
      }
    }
    imgs = root.querySelectorAll('img[src*="vehicle-images"], img[data-src*="vehicle-images"]');
    for (j = 0; j < imgs.length; j++) {
      src = imgEffectiveSrc(imgs[j]);
      if (src && isLikelyVehicleHeroPhotoUrl(src)) return src;
    }
    return '';
  }

  function digitsFromPriceText(txt) {
    if (!txt) return '';
    var n = String(txt).replace(/[^0-9.]/g, '');
    return n || '';
  }

  function parseVehicleFromClick(clickTarget) {
    var el = clickTarget && clickTarget.nodeType === 1 ? clickTarget : document.body;
    var root = findVehicleCardRoot(el);

    var vin = '';
    var stock = '';
    var price = '';
    var vehicle = '';
    var vehicleData = null;

    if (root) {
      var raw = root.getAttribute('data-vehicle');
      if (raw) {
        vehicleData = safeParseDataVehicle(raw);
        if (vehicleData && typeof vehicleData === 'object') {
          vin = String(vehicleData.vin || '');
          stock = vehicleData.stock != null ? String(vehicleData.stock) : '';
          if (vehicleData.price != null) price = String(vehicleData.price);
          else if (vehicleData.msrp != null) price = String(vehicleData.msrp);
          var parts = [vehicleData.year, vehicleData.make, vehicleData.model, vehicleData.trim].filter(function (x) {
            return x != null && String(x).trim() !== '';
          });
          if (parts.length) vehicle = parts.join(' ');
        }
      }
      if (!vin) vin = root.getAttribute('data-vehicle-vin') || '';
      if (!vehicle) {
        var titleEl = root.querySelector('h2.result-title, [data-testid="vehicle-title"]');
        if (titleEl) vehicle = titleEl.innerText.replace(/\s+/g, ' ').trim();
      }
      if (!price) {
        var pe =
          root.querySelector('.price-block.our-price.real-price .price') ||
          root.querySelector('.our-price.real-price span.price') ||
          root.querySelector('.hit-price .our-price .price');
        if (pe) price = digitsFromPriceText(pe.textContent);
      }
      if (!price) {
        var dsp = root.querySelector('[data-sales-price]');
        if (dsp) price = dsp.getAttribute('data-sales-price') || '';
      }
    }

    if (!vehicle) {
      var h1 = document.querySelector('h1');
      if (h1) vehicle = h1.innerText.replace(/\s+/g, ' ').trim();
    }
    if (!vin) {
      var vv = document.querySelector('[data-vehicle-vin]');
      if (vv) vin = vv.getAttribute('data-vehicle-vin') || '';
    }

    return { vin: vin, stock: stock, price: price, vehicle: vehicle, vehicleData: vehicleData };
  }

  function openOtpPopupFromRoot(root) {
    removePopupOverlay();

    var pageUrl = window.location.href;
    var ctx = parseVehicleFromClick(root || document.body);
    try {
      window.__otpLastUnlockCtx = { vin: ctx.vin || '', stock: ctx.stock || '' };
    } catch (e) {}
    var vin = ctx.vin;
    var stock = ctx.stock;
    var price = ctx.price;
    var vehicle = ctx.vehicle;
    var vehicleData = ctx.vehicleData;

    var overlay = document.createElement('div');
    overlay.id = 'custom-price-popup';
    overlay.style.cssText =
      'position:fixed;top:0;left:0;width:100%;height:100%;' +
      'background:linear-gradient(165deg,rgba(5,33,79,0.5) 0%,rgba(15,23,42,0.55) 100%);' +
      'backdrop-filter:saturate(115%) blur(12px);' +
      '-webkit-backdrop-filter:saturate(115%) blur(12px);' +
      'z-index:2147483647;display:flex;align-items:center;justify-content:center;';

    var box = document.createElement('div');
    box.style.cssText =
      'position:relative;width:100%;max-width:1000px;height:100vh;background:#fff;overflow:hidden;';

    var closeBtn = document.createElement('button');
    closeBtn.setAttribute('aria-label', 'Close');
    closeBtn.innerHTML = '×';
    closeBtn.style.cssText =
      'position:absolute;top:12px;right:16px;background:#fff;border:none;font-size:22px;cursor:pointer;z-index:10;width:36px;height:36px;border-radius:50%;box-shadow:0 4px 10px rgba(0,0,0,0.15);';
    closeBtn.onclick = function () {
      removePopupOverlay();
    };

    var iframe = document.createElement('iframe');
    var params =
      '?vin=' + encodeURIComponent(vin) +
      '&stock=' + encodeURIComponent(stock) +
      '&price=' + encodeURIComponent(price) +
      '&vehicle=' + encodeURIComponent(vehicle) +
      '&page_url=' + encodeURIComponent(pageUrl);

    iframe.src = POPUP_ORIGIN + '/' + params;
    iframe.style.cssText =
      'width:100%;height:100%;border:none;display:block;background:#fff;overflow:hidden;';

    iframe.addEventListener('load', function () {
      try {
        var cardRootForImg = root && root.nodeType === 1 ? root : findVehicleCardRoot(document.body);
        var heroImg = findVehicleCardHeroImageUrl(cardRootForImg || document.body);
        var snapshotBase =
          vehicleData && typeof vehicleData === 'object'
            ? Object.assign({}, vehicleData, { embed_page_url: pageUrl })
            : {
                vin: vin,
                stock: stock,
                price: price,
                vehicle_heading: vehicle,
                embed_page_url: pageUrl
              };
        if (heroImg) {
          snapshotBase = Object.assign({}, snapshotBase, {
            photoUrl: heroImg,
            imageUrl: heroImg,
            heroImage: heroImg
          });
        }
        var snapshot = snapshotBase;
        iframe.contentWindow.postMessage(
          {
            type: 'OTP_EMBED_CONTEXT',
            pageUrl: pageUrl,
            vin: vin,
            stock: stock,
            price: price,
            vehicle: vehicle,
            vehicleData: snapshot
          },
          '*'
        );
      } catch (err) {}
    });

    overlay.addEventListener('click', function (ev) {
      if (ev.target === overlay) removePopupOverlay();
    });

    box.appendChild(closeBtn);
    box.appendChild(iframe);
    overlay.appendChild(box);
    document.body.appendChild(overlay);
  }

  /** Final “dealer / our price” row — markup varies by Dealer Inspire theme */
  function findDealerPriceRow(wrap) {
    if (!wrap || !wrap.querySelector) return null;
    return (
      wrap.querySelector('.price-block.our-price.real-price') ||
      wrap.querySelector('.price-block.our-price') ||
      wrap.querySelector('.our-price.real-price') ||
      wrap.querySelector('.hit-price .our-price.real-price') ||
      wrap.querySelector('.advanced-pricing-stack .price-block.our-price') ||
      wrap.querySelector('.price-stack .price-block.our-price')
    );
  }

  /**
   * After the real dealer price is shown: restore everything we hid for suspense (stack rows, overlays)
   * but keep the legacy “Get Today’s Price” link suppressed via hideLegacyGetPriceCtasInCard.
   */
  function finalizePriceRevealOnCard(wrap) {
    if (!wrap || !wrap.querySelectorAll) return;
    wrap.classList.remove('otp-price-awaiting');
    wrap.classList.add('otp-price-revealed');
    var strayOverlays = wrap.querySelectorAll('.otp-price-lock-overlay');
    var i;
    for (i = 0; i < strayOverlays.length; i++) strayOverlays[i].remove();
    var shells = wrap.querySelectorAll('.otp-dealer-price-shell');
    for (var j = shells.length - 1; j >= 0; j--) {
      var sh = shells[j];
      var pr =
        sh.querySelector('.price-block.our-price.real-price') ||
        sh.querySelector('.price-block.our-price');
      if (pr && sh.parentNode) {
        pr.style.filter = '';
        pr.style.pointerEvents = '';
        sh.parentNode.insertBefore(pr, sh);
        sh.parentNode.removeChild(sh);
      }
    }
    var dealerRow = findDealerPriceRow(wrap);
    if (dealerRow) {
      dealerRow.style.filter = '';
      dealerRow.style.pointerEvents = '';
    }
    hideLegacyGetPriceCtasInCard(wrap);
  }

  /** Mask dealer price row + navy suspense UI (per-vehicle unlock) */
  function applyPriceLockToCard(wrap) {
    wrap.classList.remove('otp-price-awaiting');

    var row = findDealerPriceRow(wrap);
    if (!row) {
      hideLegacyGetPriceCtasInCard(wrap);
      return;
    }

    hideLegacyGetPriceCtasInCard(wrap);

    var ctx = parseVehicleFromClick(wrap);
    var ukey = unlockKeyFromContext(ctx);

    if (ukey && isKeyUnlocked(ukey)) {
      finalizePriceRevealOnCard(wrap);
      return;
    }

    wrap.classList.add('otp-price-awaiting');

    if (row.closest('.otp-dealer-price-shell')) return;

    var shell = document.createElement('div');
    shell.className = 'otp-dealer-price-shell';
    row.parentNode.insertBefore(shell, row);
    shell.appendChild(row);

    row.style.filter = 'blur(11px)';
    row.style.pointerEvents = 'none';

    var overlay = document.createElement('div');
    overlay.className = 'otp-price-lock-overlay';

    overlay.innerHTML =
      '<div class="otp-price-teaser-img-wrap">' +
        '<img class="otp-price-teaser-img" src="' +
        PRICE_TEASER_IMAGE_URL +
        '" alt="Want a Lower Price? Start here." width="640" height="200" loading="lazy" decoding="async"/>' +
      '</div>' +
      '<button type="button" class="otp-unlock-instant-btn">' +
      '<svg width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2.2" aria-hidden="true"><rect x="5" y="11" width="14" height="10" rx="2"/><path d="M7 11V8a5 5 0 0 1 10 0v3"/></svg>' +
      'Unlock Instant Price</button>';

    shell.appendChild(overlay);

    var btn = overlay.querySelector('.otp-unlock-instant-btn');
    btn.addEventListener('click', function (ev) {
      ev.preventDefault();
      ev.stopPropagation();
      if (isGloballyVerified()) {
        if (revealCardPriceWithoutPopup(wrap)) return;
      }
      openOtpPopupFromRoot(wrap);
    });
  }

  function injectPriceTeasersOnPage() {
    injectStylesOnce();
    var cards = document.querySelectorAll('.result-wrap');
    var i;
    if (cards.length) {
      for (i = 0; i < cards.length; i++) applyPriceLockToCard(cards[i]);
    } else {
      var one = document.querySelector('[data-vehicle]');
      if (one) applyPriceLockToCard(one);
    }
  }

  if (!window.__otpPopupParentBound) {
    window.__otpPopupParentBound = true;

    document.addEventListener('keydown', function (ev) {
      if (ev.key === 'Escape') removePopupOverlay();
    });

    window.addEventListener('message', function (ev) {
      if (ev.origin !== POPUP_ORIGIN) return;
      var d = ev.data;
      if (d === 'close-popup') {
        removePopupOverlay();
        return;
      }
      if (d && typeof d === 'object') {
        if (d.type === 'OTP_VEHICLE_KEYS') {
          setGloballyVerified();
          var kk = unlockKeyFromContext({ vin: d.vin, stock: d.stock });
          if (kk) addUnlockedKey(kk);
          removePopupOverlay();
          injectPriceTeasersOnPage();
          requestAnimationFrame(function () {
            injectPriceTeasersOnPage();
            guardedCelebrateVehicle(d.vin, d.stock);
          });
          return;
        }
        if (d.type === 'OTP_SUCCESS') {
          setGloballyVerified();
          var ukey =
            unlockKeyFromPayload(d.payload) ||
            unlockKeyFromContext(window.__otpLastUnlockCtx);
          if (ukey) addUnlockedKey(ukey);
          removePopupOverlay();
          injectPriceTeasersOnPage();
          requestAnimationFrame(function () {
            injectPriceTeasersOnPage();
            var vinS = '';
            var stockS = '';
            var pcar = d.payload && d.payload.car;
            if (pcar && typeof pcar === 'object') {
              vinS = String(pcar.vin || '');
              stockS = String(pcar.stock || '');
            }
            if (!vinS && !stockS) {
              try {
                var lx = window.__otpLastUnlockCtx;
                if (lx) {
                  vinS = String(lx.vin || '');
                  stockS = String(lx.stock || '');
                }
              } catch (e2) {}
            }
            guardedCelebrateVehicle(vinS, stockS);
          });
          if (window.dataLayer && Array.isArray(window.dataLayer)) {
            var p = d.payload;
            var user = p && p.user ? p.user : p;
            window.dataLayer.push({ event: 'otp_verification_success', otp_user: user, otp_car: p && p.car ? p.car : null });
          }
          return;
        }
        if (d.type === 'OTP_CLOSE') {
          removePopupOverlay();
          return;
        }
        if (d.type === 'resize') {
          var overlay = document.getElementById('custom-price-popup');
          if (!overlay) return;
          var iframe = overlay.querySelector('iframe');
          if (iframe && typeof d.height === 'number') iframe.style.height = d.height + 'px';
        }
      }
    });

    window.addEventListener('storage', function (ev) {
      if (ev.key !== VINS_STORAGE_KEY && ev.key !== GLOBAL_VERIFY_KEY) return;
      injectPriceTeasersOnPage();
    });
  }

  function injectPopup() {
    var elements = document.querySelectorAll('a, button, [class*="price"], [class*="cta"]');
    for (var i = 0; i < elements.length; i++) {
      var el = elements[i];
      if (!matchesGetTodaysPriceLabel(el.textContent || '')) continue;
      if (el.dataset.popupBound) continue;
      el.dataset.popupBound = 'true';

      var clone = el.cloneNode(true);
      el.parentNode.replaceChild(clone, el);
      clone.dataset.popupBound = 'true';

      clone.addEventListener(
        'click',
        function (e) {
          var root = findVehicleCardRoot(e.target);
          var k = unlockKeyFromContext(parseVehicleFromClick(root || e.target));
          if (k && isKeyUnlocked(k)) return;
          if (isGloballyVerified() && root && revealCardPriceWithoutPopup(root)) {
            e.preventDefault();
            e.stopPropagation();
            e.stopImmediatePropagation();
            return;
          }
          e.preventDefault();
          e.stopPropagation();
          e.stopImmediatePropagation();
          openOtpPopupFromRoot(root);
        },
        true
      );
    }
  }

  function runAll() {
    injectPopup();
    injectPriceTeasersOnPage();
  }

  var observer = new MutationObserver(function () {
    runAll();
  });
  observer.observe(document.body, { childList: true, subtree: true });

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', function () {
      setTimeout(runAll, 800);
    });
  } else {
    setTimeout(runAll, 800);
  }

  /** Custom site buttons: onclick="window.openOtpPriceEnquiryPopup(this)" or open…() with no arg */
  window.openOtpPriceEnquiryPopup = function (fromEl) {
    var root =
      fromEl && fromEl.nodeType === 1 ? findVehicleCardRoot(fromEl) : null;
    var useRoot = root || document.body;
    if (isGloballyVerified() && revealCardPriceWithoutPopup(useRoot)) return;
    openOtpPopupFromRoot(useRoot);
  };
})();

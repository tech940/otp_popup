<script>
(function () {
  var POPUP_ORIGIN = 'https://otp-popup.vercel.app';
  var DELAY_SECONDS = 10;
  var SHOW_ONLY_ON_HOME_PAGE = true;

  function removeOfferOverlay() {
    var overlay = document.getElementById('custom-offer-popup');
    if (overlay && overlay.parentNode) overlay.parentNode.removeChild(overlay);
    document.body.style.overflow = '';
  }

  function showOfferPopup() {
    // Dismissal Check
    if (sessionStorage.getItem("offer_dismissed")) return;

    removeOfferOverlay();

    var pageUrl = window.location.href;

    var overlay = document.createElement('div');
    overlay.id = 'custom-offer-popup';
    overlay.style.cssText =
      'position:fixed;top:0;left:0;width:100%;height:100%;' +
      'background:linear-gradient(165deg,rgba(5,33,79,0.5) 0%,rgba(15,23,42,0.55) 100%);' +
      'backdrop-filter:saturate(115%) blur(12px);' +
      '-webkit-backdrop-filter:saturate(115%) blur(12px);' +
      'z-index:2147483647;display:flex;align-items:center;justify-content:center;';

    var box = document.createElement('div');
    box.style.cssText =
      'position:relative;width:100%;max-width:1000px;height:100vh;background:transparent;overflow:hidden;';

    var closeBtn = document.createElement('button');
    closeBtn.setAttribute('aria-label', 'Close');
    closeBtn.innerHTML = '×';
    closeBtn.style.cssText =
      'position:absolute;top:12px;right:16px;background:#fff;border:none;font-size:22px;cursor:pointer;z-index:10;width:36px;height:36px;border-radius:50%;box-shadow:0 4px 10px rgba(0,0,0,0.15);';
    closeBtn.onclick = function () {
      removeOfferOverlay();
      sessionStorage.setItem("offer_dismissed", "true");
    };

    var iframe = document.createElement('iframe');
    var params =
      '?source=gtm-timed-offer' +
      '&page_url=' + encodeURIComponent(pageUrl) +
      '&apiBase=' + encodeURIComponent(POPUP_ORIGIN);

    iframe.src = POPUP_ORIGIN + '/offer' + params;
    iframe.style.cssText =
      'width:100%;height:100%;border:none;display:block;background:transparent;overflow:hidden;';

    box.appendChild(closeBtn);
    box.appendChild(iframe);
    overlay.appendChild(box);
    document.body.appendChild(overlay);
    document.body.style.overflow = 'hidden';

    window.addEventListener('message', function (e) {
      if (e.origin !== POPUP_ORIGIN) return;
      if ((e.data && e.data.type === 'OFFER_CLOSE') || e.data === 'close-offer') {
        removeOfferOverlay();
        sessionStorage.setItem("offer_dismissed", "true");
      }
    });
  }

  function initTimedOffer() {
    if (SHOW_ONLY_ON_HOME_PAGE) {
      var path = window.location.pathname;
      var isHome = path === "/" || path === "/index.php" || path === "" || path.endsWith("/home/");
      if (!isHome) return;
    }
    setTimeout(showOfferPopup, DELAY_SECONDS * 1000);
  }

  window.OfferPopup = { open: showOfferPopup, close: removeOfferOverlay };

  if (document.readyState === 'complete') {
    initTimedOffer();
  } else {
    window.addEventListener('load', initTimedOffer);
  }
})();
</script>

<script>
(function () {
  var POPUP_ORIGIN = 'https://am-ford-otp-popup.vercel.app';
  var ENQUIRY_KEYS = ["offer_dismissed_until","otp_verified_user","otp_enquiry_start"];

  function clearOldEnquiry() {
    ENQUIRY_KEYS.forEach(function(k){
      try { localStorage.removeItem(k); } catch(_){}
      try { sessionStorage.removeItem(k); } catch(_){}
    });
  }

  function clearEnquiryCookies() {
    var rx = /^(otp_|offer_)/;
    document.cookie.split(";").forEach(function(c) {
      var name = c.split("=")[0].trim();
      if (rx.test(name)) {
        document.cookie = name + "=;expires=Thu, 01 Jan 1970 00:00:00 GMT;path=/";
      }
    });
  }

  function getPageType() {
    var path = (window.location.pathname || '').toLowerCase();
    var isHomePath = path === '/' || path === '' || path === '/index.php' || path === '/index.html';
    var endsWithHomeSlash = path.slice(-6) === '/home/';
    var endsWithHome = path.slice(-5) === '/home';
    if (isHomePath || endsWithHomeSlash || endsWithHome) return 'Home';
    if (path.indexOf('/used-vehicles') !== -1 || path.indexOf('/new-vehicles') !== -1) return 'SRP';
    if (path.indexOf('/inventory/') !== -1) return 'VDP';
    return 'Other';
  }

  function removeTradeOfferOverlay() {
    var overlay = document.getElementById('custom-trade-offer-popup');
    if (overlay && overlay.parentNode) overlay.parentNode.removeChild(overlay);
    document.body.style.overflow = '';
  }

  function postTradeOfferContext(iframe, pageType, pageUrl) {
    if (!iframe || !iframe.contentWindow) return;
    try {
      iframe.contentWindow.postMessage({
        type: 'OTP_EMBED_CONTEXT',
        page_type: pageType,
        pageUrl: pageUrl
      }, '*');
    } catch (err) {}
  }

  function showTradeOfferPopup() {
    var pageType = getPageType();
    if (pageType !== 'Home' && pageType !== 'SRP') return;

    removeTradeOfferOverlay();

    var pageUrl = window.location.href;
    var cacheBust = new Date().getTime();
    var params =
      '?page_type=' + encodeURIComponent(pageType) +
      '&page_url=' + encodeURIComponent(pageUrl) +
      '&apiBase=' + encodeURIComponent(POPUP_ORIGIN) +
      '&_=' + cacheBust;

    var overlay = document.createElement('div');
    overlay.id = 'custom-trade-offer-popup';
    overlay.style.cssText =
      'position:fixed;top:0;left:0;width:100%;height:100%;' +
      'background:linear-gradient(165deg,rgba(5,33,79,0.52) 0%,rgba(15,23,42,0.58) 100%);' +
      'backdrop-filter:saturate(115%) blur(12px);' +
      '-webkit-backdrop-filter:saturate(115%) blur(12px);' +
      'z-index:2147483647;display:flex;align-items:center;justify-content:center;';

    var box = document.createElement('div');
    box.style.cssText =
      'position:relative;width:100%;max-width:1080px;height:100vh;' +
      'background:transparent;overflow:hidden;';

    var iframe = document.createElement('iframe');
    iframe.src = POPUP_ORIGIN + '/trade-offer' + params;
    iframe.setAttribute('allow', 'same-origin');
    iframe.style.cssText =
      'width:100%;height:100%;border:none;display:block;background:transparent;overflow:hidden;';

    iframe.addEventListener('load', function () {
      postTradeOfferContext(iframe, pageType, pageUrl);
      setTimeout(function () { postTradeOfferContext(iframe, pageType, pageUrl); }, 250);
      setTimeout(function () { postTradeOfferContext(iframe, pageType, pageUrl); }, 800);
    });

    overlay.addEventListener('click', function (ev) {
      if (ev.target === overlay) removeTradeOfferOverlay();
    });

    var escHandler = function (ev) {
      if (ev.key === 'Escape') {
        removeTradeOfferOverlay();
        document.removeEventListener('keydown', escHandler);
      }
    };
    document.addEventListener('keydown', escHandler);

    window.addEventListener('message', function (e) {
      var originCleaned = e.origin.replace(/\/$/, '');
      var targetCleaned = POPUP_ORIGIN.replace(/\/$/, '');
      if (originCleaned !== targetCleaned) return;
      if ((e.data && e.data.type === 'TRADE_OFFER_CLOSE') || e.data === 'close-trade-offer' ||
          (e.data && e.data.type === 'TRADE_OFFER_SUBMITTED')) {
        removeTradeOfferOverlay();
        document.removeEventListener('keydown', escHandler);
      }
    });

    box.appendChild(iframe);
    overlay.appendChild(box);
    document.body.appendChild(overlay);
    document.body.style.overflow = 'hidden';
  }

  function initTimedTradeOffer() {
    clearOldEnquiry();
    clearEnquiryCookies();
    var pageType = getPageType();
    if (pageType !== 'Home' && pageType !== 'SRP') return;
    var delaySeconds = 8;
    setTimeout(showTradeOfferPopup, delaySeconds * 1000);
  }

  window.TradeOfferPopup = { open: showTradeOfferPopup, close: removeTradeOfferOverlay };

  if (document.readyState === 'complete') {
    initTimedTradeOffer();
  } else {
    window.addEventListener('load', initTimedTradeOffer);
  }
})();
</script>
